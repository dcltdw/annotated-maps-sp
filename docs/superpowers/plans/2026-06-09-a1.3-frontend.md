# A1.3 — Boston Map UI (read + live viewer switcher) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** A React + MapLibre screen that renders the seeded Boston notes on a real map and re-filters markers + a note panel **live** as you switch the "viewing as" persona — the product's differentiator, made tangible in the browser, driven by the A1.2 API.

**Architecture:** Two small backend enrichments (expose each section's `rule_type` + a friendly `rule_label`; add demo-discovery endpoints for the map list and the seeded personas), then the frontend: a `MapScreen` that fetches the map + personas + visibility-filtered notes, a `MapView` (maplibre-gl markers with hover popovers + click-to-select), a self-styled `NotePanel` (sections by visibility — full vs 🔒 teaser — collapsible), and a `PreviewSwitcher`. Note **components carry their own light surface tokens**, never theme-inherited colors (the popover-contrast lesson from the mockup).

**Tech Stack:** React + TypeScript (Vite PWA), `maplibre-gl` + OpenFreeMap tiles (no key), plain `fetch`, Vitest + Testing Library. Backend: Django-Ninja (extends A1.2). Spec §5–§7; mockup approved.

**Scope guard:** READ + switcher only. **Deferred to A1.3b:** place-note create, append, edit/delete UI. No new routing (the map IS the app screen for now).

```tsx
import maplibregl from "maplibre-gl";
import { useEffect, useRef } from "react";
import "maplibre-gl/dist/maplibre-gl.css";
import type { NoteOut } from "../api/types";
import { colorFor } from "../ruleColors";

interface Props {
  center: [number, number];
  zoom: number;
  notes: NoteOut[];
  onSelect: (noteId: string) => void;
}

function peekHtml(note: NoteOut): string {
  const rows = note.sections
    .map((s) =>
      s.visibility === "teaser"
        ? `<span style="color:${colorFor(s.rule_type)}">🔒 ${s.rule_label}</span>`
        : `<span style="color:${colorFor(s.rule_type)}">● ${s.rule_label}</span> ${s.content ?? ""}`,
    )
    .join("<br>");
  return `<div style="font:13px system-ui;line-height:1.45;color:#111"><b>${note.title}</b><br>${rows}</div>`;
}

export function MapView({ center, zoom, notes, onSelect }: Props) {
  const ref = useRef<HTMLDivElement>(null);
  const mapRef = useRef<maplibregl.Map>();
  const markersRef = useRef<maplibregl.Marker[]>([]);

  useEffect(() => {
    if (!ref.current) return;
    const map = new maplibregl.Map({
      container: ref.current,
      style: "https://tiles.openfreemap.org/styles/positron",
      center,
      zoom,
    });
    map.addControl(new maplibregl.NavigationControl(), "top-right");
    mapRef.current = map;
    return () => map.remove();
    // center/zoom only seed the initial view; deps intentionally empty (the map is created once).
  }, []);

  useEffect(() => {
    const map = mapRef.current;
    if (!map) return;
    const place = () => {
      markersRef.current.forEach((m) => m.remove());
      markersRef.current = notes
        .filter((n) => n.lng != null && n.lat != null)
        .map((n) => {
          const popup = new maplibregl.Popup({ offset: 20, closeButton: false }).setHTML(peekHtml(n));
          const marker = new maplibregl.Marker({ color: "#dc2626" })
            .setLngLat([n.lng as number, n.lat as number])
            .addTo(map);
          const el = marker.getElement();
          el.style.cursor = "pointer";
          el.addEventListener("mouseenter", () => popup.setLngLat([n.lng as number, n.lat as number]).addTo(map));
          el.addEventListener("mouseleave", () => popup.remove());
          el.addEventListener("click", () => onSelect(n.id));
          return marker;
        });
    };
    // place once the style is loaded, and on every notes change
    if (map.isStyleLoaded?.()) place();
    else map.on("load", place);
  }, [notes, onSelect]);

  return <div ref={ref} className="map" data-testid="map" />;
}
```

```tsx
import { useEffect, useMemo, useState } from "react";
import { fetchMaps, fetchNotes, fetchViewers } from "./api/maps";
import type { MapOut, NoteOut, Viewer } from "./api/types";
import { MapView } from "./components/MapView";
import { NotePanel } from "./components/NotePanel";
import { PreviewSwitcher } from "./components/PreviewSwitcher";

export function MapScreen() {
  const [map, setMap] = useState<MapOut | null>(null);
  const [viewers, setViewers] = useState<Viewer[]>([]);
  const [previewAs, setPreviewAs] = useState<string | null>(null);
  const [notes, setNotes] = useState<NoteOut[]>([]);
  const [selectedId, setSelectedId] = useState<string | null>(null);
  const [panelOpen, setPanelOpen] = useState(true);

  useEffect(() => {
    fetchMaps().then((maps) => {
      const m = maps[0] ?? null;
      setMap(m);
      if (m) fetchViewers(m.id).then(setViewers);
    });
  }, []);

  useEffect(() => {
    if (map) fetchNotes(map.id, previewAs).then(setNotes);
  }, [map, previewAs]);

  const selected = useMemo(() => notes.find((n) => n.id === selectedId) ?? null, [notes, selectedId]);
  const viewerLabel = previewAs ? viewers.find((v) => v.id === previewAs)?.display_name ?? "Viewer" : "Guest";

  if (!map) return <p className="loading">Loading map…</p>;

  return (
    <div className="screen">
      <header className="topbar">
        <strong>Annotated Maps · {map.name}</strong>
        <PreviewSwitcher viewers={viewers} current={previewAs} onChange={setPreviewAs} />
      </header>
      <div className="stage">
        <div className="map-wrap">
          <MapView
            center={[map.lng, map.lat]}
            zoom={map.zoom}
            notes={notes}
            onSelect={(id) => { setSelectedId(id); setPanelOpen(true); }}
          />
          {selected && !panelOpen && (
            <button className="reopen" onClick={() => setPanelOpen(true)}>◀ note</button>
          )}
        </div>
        {selected && panelOpen && (
          <NotePanel note={selected} viewerLabel={viewerLabel} onCollapse={() => setPanelOpen(false)} />
        )}
      </div>
    </div>
  );
}
```
